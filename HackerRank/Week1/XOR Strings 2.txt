#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>




int main() {
    
    char s[1001], t[1001];  
    scanf("%s", s);
    scanf("%s", t);

    int n = strlen(s);
    char result[1001];

    for (int i = 0; i < n; i++) {
        if (s[i] == t[i])
            result[i] = '0';
        else
            result[i] = '1';
    }
    result[n] = '\0';  

    printf("%s", result);
    
    return 0;
    
}
